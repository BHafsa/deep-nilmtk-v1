        x = self.fc2(x)
        return torch.mul(sig, x)

[docs]    def forward(self, x):
        z = self.unet(x)
        out = []
        for layer in  self.out_layer:
            z_t=layer(z)
            q=(self.gate(z_t)+z_t).unsqueeze(0)
            out.append(q)
        q = torch.cat(out, 0)
        v   =  self.linear_center(z).unsqueeze(0)
        attn_output, attn_output_weights = self.multihead_attn(q, v, v)
        final_out = []
        for t, out_t in enumerate(out):
            final_out.append(F.relu(out_t + self.attn_gate(attn_output[t])))
        return torch.cat(final_out).permute(1,2,0)

        x, y  = batch
        out = self(x)
        self.q = self.q.to(x.device)
        loss = []   
        for i in range(out.size(-1)):
            pred = out[:,:,i].unsqueeze(-1)

            y_q    = y.unsqueeze(-1)[:,i].unsqueeze(0).expand_as(q_pred)
            tau      = self.q[:,None, None].expand_as(q_pred)
            loss_i = self.smooth_pinball_loss(y_q, q_pred, tau)
            loss.append(loss_i)
        loss = torch.stack(loss).sum()
    
        q_pred = torch.quantile(out, q=self.q, dim=1)
        y_q    = y.unsqueeze(0).expand_as(q_pred)
        mae = (y_q - q_pred).abs().data.sum(0).mean()
        return loss, mae

[docs]    def forward(self, x):
        z = self.unet(x).permute(0,2,1)
        pi = torch.softmax(self._pi(z), -1).permute(0,2,1)
        mu = self._mu(z).permute(0,2,1)
        log_var = self._sigma(z).permute(0,2,1)
        sigma   = elu_plus_one_plus_epsilon(log_var)
        return pi, mu, sigma

[docs]    def step(self, batch):
        x, y  = batch
        pi, mu, sigma = self(x)
        loss = []
        n_samples = []
        for i in range(pi.size(-1)):
            mix = dist.Categorical(pi[:,:,i])
            if self.dist_type =="laplace":
                comp = dist.Independent(dist.Laplace(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            elif self.dist_type=="studentT":
                comp = dist.Independent(dist.StudentT(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            else:
                comp = dist.Independent(dist.Normal(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            
            gmm = dist.MixtureSameFamily(mix, comp)
            n_samples.append(self.sample(gmm))
            
            loss_i = self.log_nlloss(y[:,i].unsqueeze(0), gmm)
            loss.append(loss_i)
        loss = torch.stack(loss).sum()
        n_samples = torch.cat(n_samples, -1)
        p10, p50, p90 = quantile(n_samples, (0.1, 0.5, 0.9))
        crps = eval_crps(n_samples, y)
        mae = (y - p50).abs().data.mean()
        return  loss, crps

        

[docs]    def predict(self,  model, test_dataloader):
        
        net = model.model.eval()
        num_batches = len(test_dataloader)
        values = range(num_batches)
        
        
        mu_probs = []
        pi_probs = []
        simga_probs = []
        true = []
        
        with tqdm(total = len(values), file=sys.stdout) as pbar:
            with torch.no_grad():
                for batch_idx, batch in enumerate(test_dataloader):
                    if len(batch)==2:
                        x, y  = batch
                        true.append(y)
                    else:
                        x  = batch
                       
                    pi, mu, sigma = net(x)
                    mu_probs.append(mu)
                    pi_probs.append(pi)
                    simga_probs.append(sigma)
                    
                    del  batch    
                    pbar.set_description('processed: %d' % (1 + batch_idx))
                    pbar.update(1)  
                
                pbar.close()
        
        mu = torch.cat(mu_probs, 0)
        sigma = torch.cat(simga_probs, 0)
        pi= torch.cat(pi_probs, 0)
        
        samples = []
        for i in range(pi.size(-1)):
            mix = dist.Categorical(pi[:,:,i])
            if self.dist_type =="laplace":
                comp = dist.Independent(dist.Laplace(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            elif self.dist_type=="studentT":
                comp = dist.Independent(dist.StudentT(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            else:
                comp = dist.Independent(dist.Normal(mu[:,:,i].unsqueeze(-1), sigma[:,:,i].unsqueeze(-1)), 1)
            
            gmm = dist.MixtureSameFamily(mix, comp)
            samples.append(self.sample(gmm))
            
        
        samples = torch.cat(samples, -1).expm1()
        p10, p50, p90 = quantile(samples, (0.1, 0.5, 0.9))

        if len(true)!=0:
            true  = torch.cat(true, 0).expm1() 
            results = {
                    "pred": p50, "sigma_pred":sigma, "mu_pred":mu, "pi_probs":pi,
                    "sample_pred":samples, "true":true}
        else:
             results = {
                    "pred": p50, "sigma_pred":sigma, "mu_pred":mu, "pi_probs":pi,
                    "sample_pred":samples}
        return results
